\chapter{حقوق دیجیتال و آزادی اینترنت}
\label{app:digital}

در دنیای مدرن، آزادی اینترنت پله‌ی اول دموکراسی است. ایران نوین دسترسی به شبکه جهانی را به عنوان یک حق بنیادین می‌شناسد.

\section{اقدامات فوری اینترنتی}

\begin{modernbox}{رفع انسداد کامل (روز اول)}
بزرگترین مانع اطلاع‌رسانی و سازماندهی در ایران، فیلترینگ است.
\begin{itemize}
    \item رفع فیلتر کامل از تمامی پلتفرم‌های بین‌المللی (تلگرام، یوتیوب، توییتر و غیره).
    \item انحلال نهادهای سانسور اینترنت و پلیس فتا به شکل فعلی.
    \item تضمین عدم قطع اینترنت (Kill Switch) تحت هیچ شرایط سیاسی.
    \item حمایت از ورود تکنولوژی‌های دسترسی ماهواره‌ای و آزاد.
\end{itemize}
\end{modernbox}

\begin{modernbox}{بلاک‌چین در خدمت دموکراسی}
طرح مهستان استفاده از تکنولوژی بلاک‌چین را نه تنها برای ارزهای دیجیتال، بلکه برای ثبت شفاف آرا، طومارهای ملی و پیگیری بودجه‌های دولتی پیشنهاد می‌دهد تا فساد به حداقل برسد.
\end{modernbox}
